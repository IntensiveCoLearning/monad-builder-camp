pragma solidity ^0.8.0;

/**
 * @title 去中心化留言板（按地址存储）
 * @notice 每个地址可以发布留言，并可通过地址查询其所有留言
 */
contract Guestbook {
    // 留言结构体：内容 + 时间戳
    struct Message {
        string content;
        uint256 timestamp;
    }

    // 每个地址 => 该地址发布的所有留言数组
    mapping(address => Message[]) private userMessages;

    // 留言添加事件，便于前端监听
    event MessageAdded(
        address indexed user,
        uint256 indexed index,
        string content,
        uint256 timestamp
    );

    /**
     * @dev 发布新留言（仅限调用者本人）
     * @param _content 留言内容（非空，长度 <= 500 字符）
     */
    function addMessage(string memory _content) public {
        require(bytes(_content).length > 0, "Content cannot be empty");
        require(bytes(_content).length <= 500, "Content too long (max 500 chars)");

        uint256 index = userMessages[msg.sender].length;
        userMessages[msg.sender].push(Message({
            content: _content,
            timestamp: block.timestamp
        }));

        emit MessageAdded(msg.sender, index, _content, block.timestamp);
    }

    /**
     * @dev 查询某个地址的留言总数
     * @param _user 要查询的地址
     * @return 留言条数
     */
    function getMessageCount(address _user) public view returns (uint256) {
        return userMessages[_user].length;
    }

    /**
     * @dev 查询某个地址的指定索引的留言
     * @param _user 要查询的地址
     * @param _index 留言索引（从 0 开始）
     * @return content 留言内容
     * @return timestamp 发布时间戳
     */
    function getMessage(address _user, uint256 _index) public view returns (string memory content, uint256 timestamp) {
        require(_index < userMessages[_user].length, "Index out of bounds");
        Message memory msg = userMessages[_user][_index];
        return (msg.content, msg.timestamp);
    }

    /**
     * @dev 获取某个地址的所有留言（返回完整数组）
     * @notice 若留言过多可能触发 gas 限制，建议使用分页查询（可参考扩展说明）
     * @param _user 要查询的地址
     * @return 该地址的所有留言数组
     */
    function getAllMessages(address _user) public view returns (Message[] memory) {
        return userMessages[_user];
    }
}

# Monad叙事

## 一、Monad 发展史

Monad 的故事始于 **2022 年 4 月**，做市巨头 Jump Trading 的研究负责人 Keone Hon 宣布离职，与两位前同事 James Hunsaker 和 Eunice Giarta 共同创立 Monad Labs。他们的核心判断是：EVM 已成为加密世界的“标配”，但以太坊主网性能严重不足，而 Layer 2 的分层策略又会破坏链上可组合性。因此，他们选择在 Layer 1 层面通过“并行处理”来改进 EVM 的执行效率。

**融资历程**贯穿了完整的市场周期：

| 轮次 | 时间 | 金额 | 领投方 |
| --- | --- | --- | --- |
| Pre-Seed | 2022年5月 | $900万 | 天使投资人、Jump Trading 网络 |
| Seed | 2023年2月 | $1,900万 | Dragonfly Capital |
| Series A | 2024年4月 | $2.25亿 | Paradigm（2024年最大加密融资） |
| 战略轮 | 2024年5月 | 未披露 | OKX Ventures 等 |
| IEO（公募） | 2025年11月 | $2.69亿 | Coinbase 平台，85,820 人参与 |

累计融资超过 **4.3 亿美元**。

**技术路线图**稳步推进：

-   **2023年9月**：官方文档发布
    
-   **2025年2月9日**：公共测试网上线，区块生成时间仅 **0.5 秒**
    
-   **2025年5月5日**：启动验证者专属 Testnet-2
    
-   **2025年11月24日**：**主网正式上线**
    

主网数据表现亮眼：总供应量 **1000 亿 MON**，向 **22.5 万名用户**进行空投。截至 2026 年 7 月，Monad TVL 已达到 **4.479 亿美元**，超越 Sui，在所有公链中位居第 **13** 位。

## 二、Monad 的优势与劣势

### ✅ 核心优势

**1\. 100% EVM 兼容（字节码级别）**

现有以太坊智能合约和生态工具可以直接迁移至 Monad，无需大幅改动。开发者可以使用 Solidity、MetaMask、Foundry 等熟悉的工具，无需学习新语言。“尽可能少动 EVM 规则”的设计哲学，使其成为“EVM 世界的并行加速器”。

**2\. 高性能与低费用**

目标性能为 **10,000 TPS**、**1 秒最终确认**、**低于 1 美分的平均 Gas 费**。Monad 通过四大技术创新实现这一目标：**乐观并行执行**（假设无冲突，同时运行所有交易，碰撞时再重新执行）、**MonadBFT 共识**（约 1 秒完成最终确认）、**异步执行**（共识与执行解耦）以及 **MonadDB**（专为高性能状态管理设计）。

**3\. 消费级硬件可运行节点**

相比 Solana 的数据中心级硬件要求，Monad 优化了节点性能，降低了硬件门槛。虽然目前仍推荐较高配置，但设计目标之一是让更多用户参与网络验证。

**4\. 顶级资本背书与生态冷启动能力**

超 **4.3 亿美元**融资储备，生态已聚集超 **280 个项目**，累计生态融资达 **13.2 亿美元**。Aave V3 上线 48 小时内存款突破 1 亿美元，Uniswap、Curve 等头部协议已部署。

### ❌ 核心劣势与挑战

**1\. 性能尚未被大规模验证**

主网上线仅半年多，**10,000 TPS 在复杂真实场景下的稳定性仍需时间验证**。市场对“性能叙事”的敏感度已明显降低——**高 TPS 早已不再是稀缺指标**。

**2\. 激励驱动的增长能否持续**

当前 TVL 和用户活跃度很大程度上由激励计划驱动。**激励退潮后能否留住用户和流动性**，是 Monad 面临的最大考验。

**3\. 代币经济学的长期压力**

MON 总供应量 **1000 亿枚**，团队（27%）和早期投资者（19.7%）持有大量代币。流通供应量仍远低于总供应量，**未来的解锁、验证者奖励、生态排放对价格表现至关重要**。MON 价格从公售价 $0.025 短暂冲高后有所回落。

**4\. 竞争格局极其激烈**

Monad 面临的不是单一对手，而是**以太坊 L2、Solana、Sui/Aptos（Move 生态）以及同赛道并行 EVM 项目（Sei、MegaETH）** 的多维度竞争。

## 三、与以太坊、Solana、Sui 的全面对比

### ⚙️ 执行模型：四者的根本差异

| 公链 | 执行模型 | 核心逻辑 |
| --- | --- | --- |
| 以太坊 | 顺序执行 | 交易像单车道上一辆接一辆处理 |
| Solana | 预先声明并行 | 交易需提前声明访问哪些账户，Sealevel 据此调度 |
| Sui | 以对象为中心（Object-Centric） | 将资产视为独立对象，拥有对象交易可绕过全局共识 |
| Monad | 乐观并行执行（Speculative EVM） | 默认无冲突，先并行执行；冲突时再按序重新执行 |

### 📊 全方位对比

| 维度 | 以太坊 | Solana | Sui | Monad |
| --- | --- | --- | --- | --- |
| 执行模型 | 顺序执行 | 预先声明并行 | 以对象为中心 | 乐观并行 EVM |
| TPS（峰值/目标） | ~15 | 数千（持续） | 亚秒级最终确认（拥有对象） | 10,000+ |
| 最终确认时间 | ~12-15 分钟 | ~400 毫秒 | <1 秒（拥有对象） | ~1 秒 |
| 虚拟机 (VM) | EVM（Solidity） | 非 EVM（Rust） | 非 EVM（Move） | 100% EVM 兼容 |
| 开发者语言 | Solidity | Rust | Move | Solidity（无需更改） |
| 验证者硬件 | 消费级 | 数据中心级 | 中等 | 数据中心级 |
| DeFi TVL（约） | 数千亿美元 | ~247.8 亿美元 | ~4.4 亿美元 | ~4.479 亿美元 |

### 🎯 各公链核心定位

**以太坊：“守成之王”**

拥有最大的开发者网络、最深的流动性和最强的网络效应。EVM 是加密领域最重要的开发者环境。但底层性能瓶颈明显，扩容依赖 L2。

**Solana：“性能之王”**

经过多轮牛熊考验的蓝筹高性能公链，在高吞吐和低费用场景中优势明显。但开发者需学习 Rust，节点硬件门槛高。

**Sui：“架构创新者”**

基于 Move 语言，以对象为中心的并行模型提供了独特优势。拥有对象交易可实现亚秒级最终确认。但生态仍在建设，Move 学习曲线对以太坊开发者不够友好。

**Monad：“EVM 的挑战者”**

核心定位可概括为 **“拥有 Solana 速度的以太坊”** 。100% EVM 兼容是最大王牌——以太坊开发者和现有应用可以**无缝迁移**，获得性能的巨大提升。Monad 选择了兼容路径，更像是 **“性能版以太坊”** ，是 EVM 世界的并行加速器。

## 💎 总结

Monad 试图在 **“性能、兼容性、去中心化”** 三者之间找到最优平衡点。它既不像 Solana 那样需要开发者学习新语言，也不像以太坊 L2 那样受限于主网的最终结算延迟。它提供了一个 **“熟悉但更快”** 的开发环境。

但挑战同样明确：**“更快”的叙事已经不够稀缺**。Monad 需要在激励退潮后证明自己拥有**不可替代的真实需求**——真实的 Gas 消耗、真实的 DeFi 活动、真实的用户留存。正如其空投后的表现所示：**空投驱动的关注度并不等同于产品与市场契合**。真正的考验，才刚刚开始。

Monad 的理解：**一个从底层软件架构出发，旨在解决“去中心化”与“高性能”对立关系的高性能 Layer-1 区块链。它试图在完全兼容以太坊生态的前提下，解决主流区块链的性能瓶颈。**

**核心技术：**  
一、**MonadBFT 共识机制**：这是Monad的共识核心，是一个前沿的拜占庭容错（BFT）协议。

-   **高性能**：它能实现**单轮（400毫秒）的推测性最终确认**和**两轮（800毫秒）的完全最终确认**。
    
-   **大规模节点支持**：在正常（无故障）路径下，其消息复杂度是线性的，允许共识节点集扩展到很大的规模。
    
-   **关键特性**：它具备**乐观响应性**（网络有多快，共识就有多快）、**领导容错**（单个失败领导者只带来一次超时延迟）以及对**尾分叉（tail-forking）攻击的抵抗能力**。
    

**二、RaptorCast**：一种高效的区块传输协议，用于优化网络层的数据分发。

**三、异步执行（Asynchronous Execution）**：这是Monad架构中非常关键的一环。它将**共识（交易排序）** 和**执行（交易计算）** 解耦。

-   **传统模式的问题**：在以太坊等链上，执行是共识的前提，导致执行时间被严重压缩，仅占区块时间的很小一部分。
    
-   **Monad的方案**：节点先对交易的**顺序**达成共识，然后再去执行它们。这使得执行可以占用**整个区块时间**，极大地提高了执行预算。
    

**四、并行执行（Parallel Execution）**：Monad 采用**乐观执行**策略。

-   **工作原理**：它会**假设交易之间没有冲突，同时开始执行一个区块内的所有交易**。
    
-   **冲突处理**：如果发生冲突（如两笔交易操作了同一个账户），系统会**重新执行**受影响的交易。由于输入数据已在缓存中，重新执行的速度远快于首次执行。最终，Monad 的执行结果与以太坊**完全相同**。
    

**五、MonadDb**：专为以太坊状态存储设计的定制化**键值数据库**。

-   **原生优化**：它原生实现了以太坊使用的**默克尔帕特里夏树（Merkle Patricia Trie）** 数据结构，避免了“一种数据结构嵌入另一种”的低效问题。
    
-   **异步I/O**：充分利用了 Linux 最新的 `io_uring` 异步I/O接口，避免因I/O操作阻塞执行。
    
-   **极致性能**：为追求极致性能，MonadDb 甚至可以**绕过文件系统，直接在块设备上运行**，以完全解锁SSD的性能潜力。
